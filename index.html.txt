    <!-- Interactive Logic and Script Integration -->
    <script>
        // Parameters for Logistic Equation
        let L = 1800;  // Asymptotic max weight (grams)
        let k = 0.28;  // Logistic growth constant
        const t0 = 12; // Inflection point (weeks)
        
        let selectedWeek = 12; // Initial state
        let chartInstance = null;

        // LaTeX formulas setup
        window.addEventListener('DOMContentLoaded', () => {
            // Initialize Lucide Icons
            lucide.createIcons();
            
            // Render Math Equations with KaTeX
            katex.render("f(t) = \\frac{L}{1 + e^{-k(t - t_0)}}", document.getElementById("formula-fx"));
            katex.render("f'(t) = k \\cdot f(t) \\left(1 - \\frac{f(t)}{L}\\right)", document.getElementById("formula-dfx"));

            // Initialize Event Listeners for Sliders
            document.getElementById('param-L').addEventListener('input', (e) => {
                L = parseFloat(e.target.value);
                document.getElementById('val-L').textContent = L + " g";
                updateSimulation();
            });

            document.getElementById('param-k').addEventListener('input', (e) => {
                k = parseFloat(e.target.value);
                document.getElementById('val-k').textContent = k.toFixed(2);
                updateSimulation();
            });

            // Initialize Visuals
            initChart();
            updateDashboard();
        });

        // Logistic Growth Function f(t)
        function logisticWeight(t) {
            return L / (1 + Math.exp(-k * (t - t0)));
        }

        // Analytical Derivative f'(t)
        function growthRate(t) {
            const w = logisticWeight(t);
            return k * w * (1 - (w / L));
        }

        // Biological Explanation Generator depending on the week/derivative values
        function getBiologicalExplanation(t, weight, rate) {
            const maxRate = k * L / 4; // Analytical maximum rate at inflection point (t=t0)
            
            if (t < 6) {
                return `<strong>Lag Phase (Fasa Perlahan)</strong>:<br>The newly induced pineapple inflorescence is mostly undergoing intense cell division but minimal cellular swelling. Biomass accumulation is slow, as reflected by the low derivative ($f'(t) = ${rate.toFixed(1)}$ g/week). Avoid high-nitrogen fertilizer shocks here; focus on maintaining soil organic matter.`;
            } else if (t >= 6 && t <= 18) {
                const closeToInflection = Math.abs(t - t0) <= 1.5;
                if (closeToInflection) {
                    return `<strong>Peak Acceleration (Fasa Infleksi)</strong>:<br>The fruit has reached its maximal growth rate. At this exact transition, the second derivative $f''(t) = 0$, meaning the derivative is at its highest value of $~${maxRate.toFixed(1)}$ g/week. This is the ultimate peak of nutrient and water demand (especially Potassium, K) to sustain structural development and build optimal brix levels!`;
                } else if (t < t0) {
                    return `<strong>Exponential Phase (Fasa Eksponen)</strong>:<br>The fruit biomass is scaling dramatically with an increasing growth rate ($f'(t)$ rising). Cells are expanding rapidly by drawing water and photosynthate. Optimal soil moisture levels are highly critical during this fast expansion window to prevent physical cracks.`;
                } else {
                    return `<strong>Decelerating Phase (Fasa Perlambatan)</strong>:<br>Although the pineapple is still gaining weight, the growth rate is slowing down (the derivative is decreasing). The fruit is shifting its metabolism from cell enlargement to sugar storage and enzyme synthesis, making organic acids equilibrate.`;
                }
            } else {
                return `<strong>Maturation & Harvest Phase (Fasa Matang)</strong>:<br>Growth has plateaued near the maximum potential weight ($L$). The growth rate derivative drops close to zero ($f'(t) = ${rate.toFixed(1)}$ g/week). Physical expansion has stopped; the crop is waiting for skin degreening (colour break index 2 to 3) before harvest for agro-processing export.`;
            }
        }

        // Core visual updater
        function updateDashboard() {
            const currentWeight = logisticWeight(selectedWeek);
            const currentRate = growthRate(selectedWeek);

            // Update UI Numbers
            document.getElementById('display-week').textContent = selectedWeek;
            document.getElementById('calc-weight').textContent = currentWeight.toFixed(1) + " g";
            document.getElementById('calc-rate').textContent = currentRate.toFixed(1) + " g/week";

            // Render Dynamic math calculation detail
            const mathElement = document.getElementById('math-der-calc');
            const mathExpression = `f'(${selectedWeek}) = ${k.toFixed(2)} \\times ${currentWeight.toFixed(1)} \\left(1 - \\frac{${currentWeight.toFixed(1)}}{${L}}\\right)`;
            katex.render(mathExpression, mathElement, { throwOnError: false });

            // Update layman physiological text
            document.getElementById('layman-explanation').innerHTML = getBiologicalExplanation(selectedWeek, currentWeight, currentRate);
        }

        // Initialize Chart.js
        function initChart() {
            const ctx = document.getElementById('growthChart').getContext('2d');
            
            const weeksArray = Array.from({length: 25}, (_, i) => i);
            const weights = weeksArray.map(w => logisticWeight(w));
            const rates = weeksArray.map(w => growthRate(w));

            chartInstance = new Chart(ctx, {
                type: 'line',
                data: {
                    labels: weeksArray,
                    datasets: [
                        {
                            label: 'Fruit Weight f(t) (g)',
                            data: weights,
                            borderColor: '#10b981', // Tailwind Emerald 500
                            backgroundColor: 'rgba(16, 185, 129, 0.1)',
                            borderWidth: 3,
                            yAxisID: 'yWeight',
                            tension: 0.35,
                            fill: true,
                            pointBackgroundColor: '#047857',
                            pointHoverRadius: 8,
                            pointHoverBackgroundColor: '#fbbf24'
                        },
                        {
                            label: "Growth Rate f'(t) (g/week)",
                            data: rates,
                            borderColor: '#f59e0b', // Tailwind Amber 500
                            borderDash: [5, 5],
                            borderWidth: 2,
                            yAxisID: 'yRate',
                            tension: 0.35,
                            pointBackgroundColor: '#d97706',
                            pointHoverRadius: 8,
                            pointHoverBackgroundColor: '#10b981'
                        }
                    ]
                },
                options: {
                    responsive: true,
                    maintainAspectRatio: false,
                    plugins: {
                        legend: {
                            display: false // Using custom styled legends above
                        },
                        tooltip: {
                            callbacks: {
                                label: function(context) {
                                    let label = context.dataset.label || '';
                                    if (label) {
                                        label += ': ';
                                    }
                                    if (context.parsed.y !== null) {
                                        label += context.parsed.y.toFixed(1);
                                    }
                                    return label;
                                }
                            }
                        }
                    },
                    // Capture clicks to identify specific graph coordinates for derivative analysis
                    onClick: (evt, activeElements) => {
                        if (activeElements.length > 0) {
                            const index = activeElements[0].index;
                            selectedWeek = index;
                            updateDashboard();
                            highlightPointOnChart(index);
                        }
                    },
                    scales: {
                        x: {
                            title: {
                                display: true,
                                text: 'Weeks Post Flower Induction (t)',
                                font: { weight: 'bold', size: 12 }
                            },
                            grid: { color: 'rgba(226, 232, 240, 0.5)' }
                        },
                        yWeight: {
                            type: 'linear',
                            position: 'left',
                            title: {
                                display: true,
                                text: 'Cumulative Weight (g)',
                                font: { weight: 'bold', size: 12, color: '#047857' }
                            },
                            min: 0,
                            max: Math.ceil(L / 100) * 100 + 100,
                            grid: { color: 'rgba(226, 232, 240, 0.5)' }
                        },
                        yRate: {
                            type: 'linear',
                            position: 'right',
                            title: {
                                display: true,
                                text: 'Biomass Gain Rate (g/week)',
                                font: { weight: 'bold', size: 12, color: '#d97706' }
                            },
                            min: 0,
                            max: Math.ceil((0.45 * 2500 / 4) / 10) * 10 + 10, // Max rate based on input ranges
                            grid: { drawOnChartArea: false } // Avoid clashing grid lines
                        }
                    }
                }
            });
        }

        // Refresh calculations and plot arrays on slider modifications
        function updateSimulation() {
            const weeksArray = Array.from({length: 25}, (_, i) => i);
            const weights = weeksArray.map(w => logisticWeight(w));
            const rates = weeksArray.map(w => growthRate(w));

            // Dynamically alter axis heights relative to parameters
            chartInstance.options.scales.yWeight.max = Math.ceil(L / 100) * 100 + 100;
            chartInstance.options.scales.yRate.max = Math.ceil((k * L / 4) / 10) * 10 + 30;

            chartInstance.data.datasets[0].data = weights;
            chartInstance.data.datasets[1].data = rates;
            
            chartInstance.update('active');
            updateDashboard();
        }

        // Mark the active week with enhanced interactive states in the chart
        function highlightPointOnChart(weekIndex) {
            // Re-render chart instance highlighting dynamic selection
            chartInstance.update();
        }
    </script>
